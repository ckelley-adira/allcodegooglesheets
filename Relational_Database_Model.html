1 Design Overview

This relational model replaces the entire 59-tab Google Sheets backend for the Adira Reads "School Lesson Tracking System." Every tab, every script-managed data store, and every hidden configuration sheet maps to a normalized PostgreSQL table on Google Cloud SQL. The design supports multi-school deployment from day one — the school_id foreign key threads through every table so the same database serves all Christel House campuses.

Design principles: Third normal form (3NF) throughout. No data duplication — the "Big Gulp" script-calculated values (skill percentages, band assignments, coaching metrics) become materialized views or computed on read, not stored redundantly. All timestamps are TIMESTAMPTZ (UTC). Soft deletes via is_active flags — no data is ever physically removed.

-- =================== CORE DOMAIN ===========================

CREATE TABLE schools (
  school_id     SERIAL PRIMARY KEY,
  name          VARCHAR(150) NOT NULL,
  short_code    VARCHAR(10)  NOT NULL UNIQUE,
  address       TEXT,
  city          VARCHAR(100),
  state         CHAR(2),
  is_active     BOOLEAN NOT NULL DEFAULT TRUE,
  created_at    TIMESTAMPTZ NOT NULL DEFAULT now(),
  updated_at    TIMESTAMPTZ NOT NULL DEFAULT now()
);

CREATE TABLE academic_years (
  year_id       SERIAL PRIMARY KEY,
  school_id     INT NOT NULL REFERENCES schools(school_id),
  label         VARCHAR(10) NOT NULL,  -- e.g. 'FY26'
  start_date    DATE NOT NULL,
  end_date      DATE NOT NULL,
  is_current    BOOLEAN NOT NULL DEFAULT FALSE,
  UNIQUE(school_id, label)
);

CREATE TABLE grade_levels (
  grade_id      SERIAL PRIMARY KEY,
  name          VARCHAR(5) NOT NULL UNIQUE,  -- 'KG','G1'...'G8'
  sort_order    SMALLINT NOT NULL,
  grade_band    VARCHAR(10) NOT NULL  -- 'K-2', '3-5', '6-8'
);

CREATE TABLE teachers (
  teacher_id    SERIAL PRIMARY KEY,
  school_id     INT NOT NULL REFERENCES schools(school_id),
  first_name    VARCHAR(80) NOT NULL,
  last_name     VARCHAR(80) NOT NULL,
  email         VARCHAR(200) NOT NULL UNIQUE,
  role          teacher_role NOT NULL DEFAULT 'teacher',
  is_active     BOOLEAN NOT NULL DEFAULT TRUE,
  created_at    TIMESTAMPTZ NOT NULL DEFAULT now(),
  updated_at    TIMESTAMPTZ NOT NULL DEFAULT now()
);

CREATE TABLE students (
  student_id        SERIAL PRIMARY KEY,
  school_id         INT NOT NULL REFERENCES schools(school_id),
  grade_id          INT NOT NULL REFERENCES grade_levels(grade_id),
  first_name        VARCHAR(80) NOT NULL,
  last_name         VARCHAR(80) NOT NULL,
  student_number    VARCHAR(20) NOT NULL UNIQUE,
  enrollment_status enrollment_status NOT NULL DEFAULT 'active',
  enrollment_date   DATE NOT NULL,
  withdrawal_date   DATE,
  created_at        TIMESTAMPTZ NOT NULL DEFAULT now(),
  updated_at        TIMESTAMPTZ NOT NULL DEFAULT now()
);

CREATE TABLE instructional_groups (
  group_id      SERIAL PRIMARY KEY,
  school_id     INT NOT NULL REFERENCES schools(school_id),
  grade_id      INT NOT NULL REFERENCES grade_levels(grade_id),
  year_id       INT NOT NULL REFERENCES academic_years(year_id),
  teacher_id    INT NOT NULL REFERENCES teachers(teacher_id),
  group_name    VARCHAR(100) NOT NULL,
  is_mixed_grade BOOLEAN NOT NULL DEFAULT FALSE,
  is_active     BOOLEAN NOT NULL DEFAULT TRUE,
  created_at    TIMESTAMPTZ NOT NULL DEFAULT now(),
  updated_at    TIMESTAMPTZ NOT NULL DEFAULT now(),
  UNIQUE(school_id, year_id, group_name)
);

CREATE TABLE group_memberships (
  membership_id SERIAL PRIMARY KEY,
  group_id      INT NOT NULL REFERENCES instructional_groups(group_id),
  student_id    INT NOT NULL REFERENCES students(student_id),
  joined_date   DATE NOT NULL DEFAULT CURRENT_DATE,
  left_date     DATE,
  is_active     BOOLEAN NOT NULL DEFAULT TRUE,
  UNIQUE(group_id, student_id, joined_date)
);

-- =================== UPDATED_AT TRIGGER ======================

CREATE OR REPLACE FUNCTION update_updated_at()
RETURNS TRIGGER AS $$
BEGIN
  NEW.updated_at = now();
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- Apply to all tables with updated_at columns
CREATE TRIGGER trg_schools_updated
  BEFORE UPDATE ON schools
  FOR EACH ROW EXECUTE FUNCTION update_updated_at();

CREATE TRIGGER trg_teachers_updated
  BEFORE UPDATE ON teachers
  FOR EACH ROW EXECUTE FUNCTION update_updated_at();

CREATE TRIGGER trg_students_updated
  BEFORE UPDATE ON students
  FOR EACH ROW EXECUTE FUNCTION update_updated_at();

CREATE TRIGGER trg_groups_updated
  BEFORE UPDATE ON instructional_groups
  FOR EACH ROW EXECUTE FUNCTION update_updated_at();

4 Indexing Strategy

These indexes target the actual query patterns the Adira Reads system runs — report generation, student lookups, weekly coaching metrics, and band movement analysis.

-- =================== PERFORMANCE INDEXES ====================

-- lesson_progress: the largest table — every query pattern needs coverage
CREATE INDEX idx_progress_student_year
  ON lesson_progress(student_id, year_id);
CREATE INDEX idx_progress_group_date
  ON lesson_progress(group_id, date_recorded);
CREATE INDEX idx_progress_lesson_status
  ON lesson_progress(lesson_id, status);
CREATE INDEX idx_progress_year_status
  ON lesson_progress(year_id, status)
  WHERE status = 'Y';  -- partial index for pass-rate queries

-- students: filtered by school, grade, enrollment status
CREATE INDEX idx_students_school_grade
  ON students(school_id, grade_id)
  WHERE enrollment_status = 'active';

-- group_memberships: join table queried constantly
CREATE INDEX idx_memberships_student
  ON group_memberships(student_id)
  WHERE is_active = TRUE;
CREATE INDEX idx_memberships_group
  ON group_memberships(group_id)
  WHERE is_active = TRUE;

-- weekly_snapshots: time-series queries
CREATE INDEX idx_snapshots_student_week
  ON weekly_snapshots(student_id, year_id, week_number);

-- coaching_metrics: Friday Dashboard lookups
CREATE INDEX idx_coaching_group_week
  ON coaching_metrics(group_id, week_start_date DESC);
CREATE INDEX idx_coaching_priority
  ON coaching_metrics(coaching_priority, year_id)
  WHERE coaching_priority IN ('critical', 'high');

-- benchmark_records: band movement queries
CREATE INDEX idx_benchmarks_student_year
  ON benchmark_records(student_id, year_id);

-- assessment_results: per-student assessment lookups
CREATE INDEX idx_assessments_student_year
  ON assessment_results(student_id, year_id);

-- tutoring_sessions: dual-track queries
CREATE INDEX idx_tutoring_student_year
  ON tutoring_sessions(student_id, year_id);

-- audit_log: admin queries by school and date
CREATE INDEX idx_audit_school_date
  ON audit_log(school_id, created_at DESC);
Why partial indexes? The WHERE status = 'Y' partial index on lesson_progress means pass-rate calculations only scan rows that actually passed — not the full table. With 405 students × 128 lessons = ~51,840 rows per year (and growing with multi-school), this matters. The WHERE is_active = TRUE filters on group_memberships and students mean current-state queries skip historical records entirely.

6 Migration Plan: Sheets → Cloud SQL

Phase 1

Infrastructure Setup (Week 1)

Provision Cloud SQL instance: PostgreSQL 15, db-custom-2-8192 (2 vCPU, 8GB RAM) is sufficient for the current 405-student, single-school workload. Enable automated backups and point-in-time recovery. Region: us-central1 (closest to Indianapolis).

Networking: Private IP within a VPC. Set up Cloud SQL Auth Proxy for secure local connections during migration. If an AppSheet front-end is planned, enable public IP with SSL enforcement + authorized networks.

Run the DDL: Execute the full schema (Section 3) and indexes (Section 4) against the new instance. Validate with \dt and \di to confirm all 22 tables and indexes are created.

Critical data mapping: The current system uses student names as identifiers in group sheets. The database uses student_id. The migration script must fuzzy-match student names from group sheets to the Student Roster to establish correct foreign keys. The studentrecon.gs script already does something similar — reuse its matching logic.
Phase 4

Lesson Progress Migration (Week 3–4)

This is the biggest transformation. The UFLI Map tab stores lesson progress as a wide table: one row per student, 128 columns (one per lesson), with Y/N/A values. The database stores this as a narrow table: one row per student-lesson pair.

Unpivot strategy:

  1. Export the UFLI Map tab as CSV (Apps Script → Drive → download)
  2. Write a Python script that reads the wide CSV and unpivots: for each student row, iterate columns 1–128, emit a (student_id, lesson_id, status, date_recorded, source='import') row
  3. Bulk load via COPY — expect ~51,840 rows (405 students × 128 lessons)
  4. Validate: SELECT COUNT(*) FROM lesson_progress WHERE source = 'import' should match expected total

Date handling: The Sheets system doesn't store individual lesson dates in the UFLI Map — it only has the current status. For imported data, set date_recorded = date of migration and source = 'import'. Going forward, the form-based entry will capture actual dates.

7 Key Views & Queries

These views replace the script-calculated dashboard tabs. They're computed from live data — no stale caches, no "Big Gulp" batch needed.

School Summary View (replaces School Summary tab)

CREATE MATERIALIZED VIEW mv_school_summary AS
WITH student_progress AS (
  SELECT
    s.student_id,
    s.school_id,
    g.name AS grade_name,
    g.sort_order,
    COUNT(lp.progress_id) FILTER (WHERE lp.status = 'Y') AS passed,
    COUNT(lp.progress_id) FILTER (WHERE lp.status IN ('Y','N')) AS attempted,
    ROUND(
      100.0 * COUNT(lp.progress_id) FILTER (WHERE lp.status = 'Y')
      / NULLIF(COUNT(lp.progress_id) FILTER (WHERE lp.status IN ('Y','N')), 0),
    1) AS current_pct
  FROM students s
  JOIN grade_levels g ON g.grade_id = s.grade_id
  LEFT JOIN lesson_progress lp ON lp.student_id = s.student_id
  WHERE s.enrollment_status = 'active'
  GROUP BY s.student_id, s.school_id, g.name, g.sort_order
)
SELECT
  school_id,
  grade_name,
  sort_order,
  COUNT(*) AS total_students,
  COUNT(*) FILTER (WHERE current_pct >= 80) AS on_track,
  COUNT(*) FILTER (WHERE current_pct >= 50 AND current_pct < 80) AS progressing,
  COUNT(*) FILTER (WHERE current_pct < 50) AS needs_support,
  ROUND(AVG(current_pct), 1) AS avg_current_pct
FROM student_progress
GROUP BY school_id, grade_name, sort_order
ORDER BY sort_order;

Band Movement View (replaces Band Movement Analysis tab)

CREATE VIEW v_band_movement AS
SELECT
  s.student_id,
  s.first_name || ' ' || s.last_name AS student_name,
  g.name AS grade,
  bi.band_label AS initial_band,
  bi.percentage AS initial_pct,
  bc.band_label AS current_band,
  bc.percentage AS current_pct,
  bc.percentage - bi.percentage AS growth,
  CASE
    WHEN bc.band_label > bi.band_label THEN 'improved'
    WHEN bc.band_label = bi.band_label THEN 'stable'
    ELSE 'declined'
  END AS movement
FROM students s
JOIN grade_levels g ON g.grade_id = s.grade_id
LEFT JOIN benchmark_records bi
  ON bi.student_id = s.student_id AND bi.benchmark_type = 'initial'
LEFT JOIN benchmark_records bc
  ON bc.student_id = s.student_id AND bc.benchmark_type = 'current'
WHERE s.enrollment_status = 'active';

Group Progress View (replaces all 24 Group Sheet tabs)

-- One view replaces KG Groups, G1 Groups, G2 Groups, ... (×24 tabs)
CREATE VIEW v_group_progress AS
SELECT
  ig.group_name,
  g.name AS grade,
  s.first_name || ' ' || s.last_name AS student_name,
  ul.lesson_number,
  ul.skill_section,
  lp.status,
  lp.date_recorded
FROM instructional_groups ig
JOIN group_memberships gm ON gm.group_id = ig.group_id AND gm.is_active = TRUE
JOIN students s ON s.student_id = gm.student_id
JOIN grade_levels g ON g.grade_id = s.grade_id
LEFT JOIN lesson_progress lp ON lp.student_id = s.student_id AND lp.group_id = ig.group_id
LEFT JOIN ufli_lessons ul ON ul.lesson_id = lp.lesson_id
WHERE ig.is_active = TRUE
ORDER BY ig.group_name, s.last_name, ul.lesson_number;

Skills Tracker View (replaces Skills Tracker tab)

CREATE VIEW v_skills_tracker AS
SELECT
  s.student_id,
  s.first_name || ' ' || s.last_name AS student_name,
  g.name AS grade,
  ul.skill_section,
  COUNT(*) FILTER (WHERE lp.status IN ('Y','N')) AS lessons_taken,
  COUNT(*) FILTER (WHERE lp.status = 'Y') AS lessons_passed,
  ROUND(
    100.0 * COUNT(*) FILTER (WHERE lp.status = 'Y')
    / NULLIF(COUNT(*) FILTER (WHERE lp.status IN ('Y','N')), 0),
  1) AS pass_pct
FROM students s
JOIN grade_levels g ON g.grade_id = s.grade_id
JOIN lesson_progress lp ON lp.student_id = s.student_id
JOIN ufli_lessons ul ON ul.lesson_id = lp.lesson_id
WHERE s.enrollment_status = 'active'
GROUP BY s.student_id, s.first_name, s.last_name, g.name, ul.skill_section
ORDER BY g.name, s.last_name, ul.skill_section;
